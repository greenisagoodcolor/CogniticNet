### Subtask 1.1: Create Complete File Inventory
**Technical Steps:**
1. Traverse entire repository using `os.walk()` or `pathlib`
2. Calculate MD5 hash for each file
3. Extract file metadata (size, last modified, permissions)
4. Store in structured format (JSON/SQLite)

**Implementation:**
```python
def inventory_file(path: Path) -> dict:
    """
    Create inventory entry for a file with comprehensive metadata.

    This function captures everything needed to track file movements
    and detect corruption during reorganization.

    Args:
        path: Path to the file to inventory

    Returns:
        dict: File metadata including hash, size, and timestamps
    """
    return {
        'path': str(path),
        'hash': hashlib.md5(path.read_bytes()).hexdigest(),
        'size': path.stat().st_size,
        'modified': path.stat().st_mtime,
        'type': identify_file_type(path)
    }
```

**Rationale (Feathers):** "In legacy code, you must understand the current state completely before making changes. The hash allows us to detect duplicates and verify nothing was corrupted during reorganization."

**Implementation:**
```python
def extract_python_dependencies(file_path: Path) -> Set[str]:
    """
    Extract all dependencies from a Python file using AST parsing.

    Handles both static imports (import X, from X import Y) and
    dynamic imports (__import__, importlib).

    Args:
        file_path: Path to Python file to analyze

    Returns:
        Set[str]: All module names this file depends on
    """
    tree = ast.parse(file_path.read_text())
    imports = set()

    for node in ast.walk(tree):
        if isinstance(node, ast.Import):
            imports.update(alias.name for alias in node.names)
        elif isinstance(node, ast.ImportFrom):
            if node.module:
                imports.add(node.module)

    # Also check dynamic imports
    content = file_path.read_text()
    dynamic = re.findall(r'__import__\(["\']([^"\']+)', content)
    imports.update(dynamic)

    return imports
```

**Rationale (Sandi Metz):** "Dependencies are the key to understanding change propagation. A file with many dependents is risky to move - you must update all import paths."

    """
    Classify a Python file based on its content, not just its name.

    This deep inspection ensures files are placed according to their
    actual purpose, not what their name suggests.

    Args:
        path: Path to Python file to classify

    Returns:
        str: Classification category
    """
    content = path.read_text()

    # Order matters - check most specific first
    if 'test_' in path.name or 'pytest' in content:
        return 'test'
    elif '@router' in content or 'APIRouter' in content:
        return 'api_route'
    elif 'Base' in content and 'Column' in content:
        return 'database_model'
    elif 'Schema' in content and 'pydantic' in content:
        return 'api_schema'
    elif 'Service' in content or 'service' in path.name:
        return 'service'
    elif 'def ' in content and 'class' not in content:
        return 'utility'
    else:
        return 'unknown'

def plan_movements(files: List[Path]) -> List[Move]:
    """
    Plan file movements respecting dependency order.

    Files with no dependencies move first, then their dependents.
    This ensures imports remain valid throughout the process.
    """
    moves = []

    sorted_files = topological_sort(files, dependency_graph)

def update_imports(file_path: Path, import_changes: Dict[str, str]):
    """
    Update import statements using AST manipulation for safety.

    String replacement would change imports in strings/comments.
    AST rewriting only changes actual import statements.
    """
    tree = ast.parse(file_path.read_text())

    class ImportUpdater(ast.NodeTransformer):
        def visit_Import(self, node):
            for alias in node.names:
                if alias.name in import_changes:
                    alias.name = import_changes[alias.name]
            return node

        def visit_ImportFrom(self, node):
            if node.module in import_changes:
                node.module = import_changes[node.module]
            return node

    updated_tree = ImportUpdater().visit(tree)
    file_path.write_text(ast.unparse(updated_tree))
```

**Rationale (Metz):** "String replacement for imports is dangerous. AST rewriting ensures we only change actual imports, not strings that look like imports."

def execute_batch(moves: List[Move], batch_size: int = 10):
    """
    Execute file movements in small, tested batches.

    Each batch is validated before proceeding. On failure,
    the batch is rolled back to maintain consistency.
    """
    for i in range(0, len(moves), batch_size):
        batch = moves[i:i + batch_size]

        # Execute moves
        for move in batch:
            shutil.move(move.source, move.target)
            update_imports_for_move(move)

        # Validate
        if not run_critical_tests():
            rollback_batch(batch)
            raise Exception(f"Batch {i//batch_size} failed tests")

        # Commit
        git_commit(f"refactor: Reorganize batch {i//batch_size}")
```

import numpy as np

class TestActiveInferenceInvariants:
    """
    Property-based tests ensuring Active Inference mathematical invariants hold.
    """

    @given(
        beliefs=st.lists(
            st.floats(min_value=0, max_value=1),
            min_size=2,
            max_size=10
        )

    def test_beliefs_sum_to_one(self, beliefs):
        """Belief distributions must always sum to 1 (probability constraint)."""
        belief_array = np.array(beliefs)
        normalized = belief_array / belief_array.sum()

        agent = Agent.create("Explorer")
        agent.beliefs.update(normalized)

        assert abs(agent.beliefs.sum() - 1.0) < 1e-10

    @given(
        num_agents=st.integers(min_value=2, max_value=10),
        timesteps=st.integers(min_value=10, max_value=100)
    )
    def test_coalition_formation_increases_value(self, num_agents, timesteps):
        """
        Coalition formation should only occur when it increases expected value
        for all participants (Pareto improvement).
        """
        agents = [Agent.create("Explorer") for _ in range(num_agents)]
        world = World()

        for agent in agents:
            world.add_agent(agent)
            agent.start_value = agent.calculate_expected_value()

        # Run simulation
        for _ in range(timesteps):
            world.step()

        # Check all coalitions
        for coalition in world.coalitions:
            for agent_id in coalition.members:
                agent = world.get_agent(agent_id)
                assert agent.calculate_expected_value() >= agent.start_value

        likelihood: np.ndarray,    # Shape: (n_observations, n_states)
        prior: np.ndarray         # Shape: (n_states,)
